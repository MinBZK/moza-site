{{- /* Render een link met automatische detectie van externe links. Parameters (dict): - href: De URL - text: De linktekst - title: (optioneel) Title attribuut Externe links krijgen automatisch target="_blank" en rel="external noopener noreferrer". Links naar samenwerkruimten.nl krijgen een "(besloten omgeving)" label. */ -}} {{- $u := urls.Parse .href -}} {{- $isExternal := and $u.IsAbs (hasPrefix $u.Scheme "http") -}} {{- $isPrivate := and $isExternal (strings.Contains $u.Host "samenwerkruimten.nl") -}} {{- if $isPrivate -}} {{ .text }} (besloten omgeving) {{- else if $isExternal -}} {{ .text }} {{- else -}} {{ .text }} {{- end -}}
